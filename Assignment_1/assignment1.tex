\textbf{Question:} A box contains 5 red marbles, 8 white marbles and 4 green marbles.One marble is taken
out of the box at random. What is the probability that the marble taken out will be :

\begin{enumerate}[label=(\roman*)]
    \item red ? 
    \item white ? 
    \item  not green?
\end{enumerate}


\textbf{Solution :}


 Number of red marbles = 5


 Number of white marbles = 8


 Number of green marbles = 4


  Total marbles = 5+8+4 = 17

\begin{table}[ht!]
\centering
  
\input{table/table.tex}

% \caption{}
% \label{table}

\end{table}

    
\begin{enumerate}[label=(\roman*)]
\item Probability that the marble taken out is red

\begin{align}
&=\frac{\text{ Number of red marbles}}{\text{ Total number of marbles}}
\end{align}
    
% = $\frac{\text{\large Number of red marbles}}{\text{\large Total number of marbles}}\quad$

Probability that the marble taken out is red:

\begin{align}
	\pr{X=0} &= \frac{n(X=0)}{\sum_{i=0}^2 n(X=i)} = \frac{5}{17} \\
	\therefore \pr{X=0} &= \frac{5}{17} \approx 0.2941
	\end{align}

% = $\frac{\text{\large 5}}{\text{\large 17}}\quad$


\item Probability that the marble taken out is white

\begin{align}
&=\frac{\text{ Number of white marbles}}{\text{ Total number of marbles}}
\end{align}
% = $\frac{\text{\large Number of white marbles}}{\text{\large Total number of marbles}}\quad$


Probability that the marble taken out is white:

\begin{align}
	\pr{X=1} &= \frac{n(X=1)}{\sum_{i=0}^2 n(X=i)} = \frac{8}{17} \\
	\therefore \pr{X=1} &= \frac{8}{17} \approx 0.4706
	\end{align}

% = $\frac{\text{\large 8}}{\text{\large 17}}\quad$


\item Probability that the marble taken out is not green


\begin{align}
&=\frac{\text{ Number of non green marbles}}{\text{ Total number of marbles}}
\end{align}


% = $\frac{\text{\large Number of non green marbles}}{\text{\large Total number of marbles}}\quad$


Probability that the marble taken out is not green:


\begin{align}
	\pr{X \ne 2} &= \frac{n(X \ne 2)}{\sum_{i=0}^2 n(X=i)} = \frac{13}{17} \\
	\therefore \pr{X \ne 2} &= \frac{13}{17} \approx 0.7647
	\end{align}

% = $\frac{\text{\large 13}}{\text{\large 17}}\quad$
\end{enumerate}
